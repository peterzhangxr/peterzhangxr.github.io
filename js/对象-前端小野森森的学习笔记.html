# 包装类

访问属性的时候对基本类型做一次包装

var a = 123
a.len = 3
//new Number(123).len = 3;delete
console.log(a.len) // undefined

var a = 'abc'
console.log(a.length) //3
var arr = [1, 2, 3, 4, 5]
arr.length = 3
console.log(arr)//[1,2,3]
